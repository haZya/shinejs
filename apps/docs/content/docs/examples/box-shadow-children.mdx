---
title: Box Shadow with Children
description: Apply shinejs to child-rich layouts like cards and images.
---

This pattern mirrors `apps/demo/src/app/auto-pilot-children/page.tsx`.

<PreviewCodeTabs
  language="tsx"
  code={`"use client";

import { useRef } from "react";
import { useShine } from "shinejs/react";

export function ImageGridShadow() {
  const ref = useRef<HTMLDivElement>(null);

  useShine(ref, {
    config: {
      shadowRGB: { r: 0, g: 0, b: 0 },
      blur: 22,
      opacity: 0.2,
    },
  });

  return (
    <div ref={ref} className="grid grid-cols-3 gap-4">
      <img src="/img-1.jpg" alt="1" />
      <img src="/img-2.jpg" alt="2" />
      <img src="/img-3.jpg" alt="3" />
    </div>
  );
}
`}

</PreviewCodeTabs>

Because the target has child elements, shinejs uses `boxShadow` automatically.
