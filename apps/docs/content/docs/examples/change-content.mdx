---
title: Change Content
description: Replace target text at runtime with update({ content }).
---

Recreate the behavior from [`apps/demo/src/app/change-text/page.tsx`](https://github.com/haZya/shinejs/tree/6d200502eb25859aea4bcf17b2cd6f4340d2c1fa/apps/demo/src/app/change-text).

<PreviewCodeTabs
  language="tsx"
  code={`"use client";

  import { useRef } from "react";
  import { useShine } from "@hazya/shinejs/react";

  export function ContentSwap() {
    const ref = useRef<HTMLHeadingElement>(null);
    const { update } = useShine(ref);

    return (
      <>
        <button onClick={() => update({ content: "Change Text Demo" })}>
          Change Text Demo
        </button>
        <button onClick={() => update({ content: "Hello World" })}>
          Hello World
        </button>
        <button onClick={() => update({ content: "Lorem Ipsum" })}>
          Lorem Ipsum
        </button>
        <h1 ref={ref}>Change Text Demo</h1>
      </>
    );
  }
`}
