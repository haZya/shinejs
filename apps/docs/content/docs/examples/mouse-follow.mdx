This is the simplest interaction model and mirrors the demo route [`apps/demo/src/app/mouse-follow/page.tsx`](https://github.com/haZya/shinejs/tree/6d200502eb25859aea4bcf17b2cd6f4340d2c1fa/apps/demo/src/app/mouse-follow).

<PreviewCodeTabs
  language="tsx"
  code={`"use client";

  import { useRef } from "react";
  import { useShine } from "@hazya/shinejs/react";

  export function MouseFollowHeadline() {
    const ref = useRef<HTMLHeadingElement>(null);
    useShine(ref, { light: { position: "followMouse" } });

    return <h1 ref={ref}>Shine Mouse Follow</h1>;
  }
`}
