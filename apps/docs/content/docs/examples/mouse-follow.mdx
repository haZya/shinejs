---
title: Mouse Follow
description: Canonical cursor-follow shadow behavior.
---

This is the simplest interaction model and mirrors the demo route `apps/demo/src/app/mouse-follow/page.tsx`.

```tsx
"use client";

import { useRef } from "react";
import { useShine } from "shinejs/react";

export function MouseFollowHeadline() {
  const ref = useRef<HTMLHeadingElement>(null);
  useShine(ref, {
    light: { position: "followMouse" },
    config: { shadowRGB: { r: 0, g: 0, b: 0 } },
  });

  return <h1 ref={ref}>Shine Mouse Follow</h1>;
}
```
