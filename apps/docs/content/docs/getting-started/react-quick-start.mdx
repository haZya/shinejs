## Overview

`useShine` is the fastest way to apply interactive shadows in React.

<MouseFollowPreview />

## Usage

```tsx
"use client";

import { useRef } from "react";
import { useShine } from "shinejs/react";

export function HeroTitle() {
  const titleRef = useRef<HTMLHeadingElement>(null);

  useShine(titleRef, {
    light: { position: "followMouse" },
    config: {
      numSteps: 8,
      opacity: 0.3,
      blur: 40,
    },
  });

  return <h1 ref={titleRef}>Shining Bright</h1>;
}
```
