## Overview

[`useShine`](/docs/guides/react-use-shine) is the fastest way to apply interactive shadows in React.

## Basic Usage

```tsx
"use client";

import { useRef } from "react";
import { useShine } from "shinejs/react";

export function HeroTitle() {
  const titleRef = useRef<HTMLHeadingElement>(null);
  
  useShine(titleRef, { light: { position: "followMouse" } });

  return <h1 ref={titleRef}>Shining Bright</h1>;
}
```
